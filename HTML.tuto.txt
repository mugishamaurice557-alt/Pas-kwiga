<script>
const questions = [
    {
        question: "Ijambo 'umunyeshuri' risobanura iki?",
        options: ["Uwigisha", "Uwiga", "Umuyobozi", "Umubyeyi"],
        answer: 1
    },
    {
        question: "Ni irihe jambo riri mu cyiciro cy'inshinga?",
        options: ["Inzu", "Gusoma", "Umwana", "Igitabo"],
        answer: 1
    },
    {
        question: "Ijambo 'amahoro' risobanura iki?",
        options: ["Intambara", "Umutekano", "Uburwayi", "Inzara"],
        answer: 1
    }
];

let currentQuestion = 0;
let score = 0;
let timeLeft = 15;
let timer;
function startTimer() {
    timeLeft = 15;
    document.getElementById("timer").innerText = "Igihe: " + timeLeft;
    timer = setInterval(() => {
        timeLeft--;
        document.getElementById("timer").innerText = "Igihe: " + timeLeft;
        if (timeLeft === 0) {
            clearInterval(timer);
            nextQuestion();
        }
    }, 1000);
}
function loadQuestion() {
    clearInterval(timer);
    st
    const q = questions[currentQuestion];
    document.getElementById("question").innerText = q.question;
    const optionsDiv = document.getElementById("options");
    optionsDiv.innerHTML = "";
    q.options.forEach((option, index) => {
        const div = document.createElement("div");
        div.classList.add("option");
        div.innerText = option;
        div.onclick = () => selectAnswer(div, index);
        optionsDiv.appendChild(div);
    });
}

function selectAnswer(element, index) {
    const q = questions[currentQuestion];
    const options = document.querySelectorAll(".option");

    options.forEach(opt => opt.onclick = null);

    if (index === q.answer) {
        element.classList.add("correct");
        score++;
    } else {
        element.classList.add("wrong");
        options[q.answer].classList.add("correct");
    }
}

function nextQuestion() {
    currentQuestion++;
    if (currentQuestion < questions.length) {
        loadQuestion();
    } else {
        showResult();
    }
}
function showResult() {
    clearInterval(timer);
    document.querySelector(".container").innerHTML = `
        <h1>Byarangiye</h1>
        <p>Amanota yawe: ${score} / ${questions.length}</p>
        <button onclick="location.reload()">Ongera utangire</button>
    `;
}
document.getElementById("nextBtn").addEventListener("click", nextQuestion);
loadQuestion();
</script>
